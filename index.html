Installation

npm install devmode-wc

Basic Usage

<!-- Using from CDN -->
<script type="module" src="https://unpkg.com/devmode-wc/dist/devmode-wc/devmode-wc.esm.js"></script>

<!-- Basic usage -->
<devmode-wc></devmode-wc>

<!-- With single origin -->
<devmode-wc allowed-origins="http://localhost:3000"></devmode-wc>

<!-- With multiple origins -->
<devmode-wc allowed-origins="http://localhost:3000, https://arkits.github.io"></devmode-wc>

<!-- With event listener -->
<devmode-wc onPillToggled="(e) => console.log(e.detail)"></devmode-wc>
